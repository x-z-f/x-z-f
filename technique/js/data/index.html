# new Date()转YYYY-MM-DD HH:mm:ss

/**
 * @description new Date （国际时间） 转化为YYYY-MM-DD HH:mm:ss
 * @param date{Date}
 * @return {String} 返回string类型
 */
export function dateToFormat(date) {
  if (date instanceof Date) {
    return date
      .toLocaleDateString("en-US", { hour12: false })
      .replace(/\b\d\b/g, "0$&")
      .replace(new RegExp("/", "gm"), "-");
  } else {
    console.error("not Date Type");
  }
}

# 获取当前的时间yyyy-MM-dd HH:mm:ss

/**
 * @function 获取当前的时间yyyy-MM-dd HH:mm:ss
 * @description:  没有满10就补0
 * @param
 * @return {String} 返回String类型
 */
export function obtainDate() {
  let date = new Date();
  let year = date.getFullYear();
  let month = date.getMonth() + 1;
  let day = date.getDate();
  let hours = date.getHours();
  let minu = date.getMinutes();
  let second = date.getSeconds();
  //判断是否满10
  let arr = [month, day, hours, minu, second];
  arr.forEach((item) => {
    item < 10 ? "0" + item : item;
  });
  return (
    year +
    "-" +
    arr[0] +
    "-" +
    arr[1] +
    " " +
    arr[2] +
    ":" +
    arr[3] +
    ":" +
    arr[4]
  );
}

# 计算两个日期格式为(yyyy-MM-dd)相差几个月

/**
 * @function 计算两个日期格式为(yyyy-MM-dd)相差几个月
 * @description:
 * @param dateOne{Date}
 * @param dateTwo{Date}
 * @return String
 */
export function disparityFewMonth(dateOne, dateTwo) {
  let datesOne = dateOne.split("-").map((item) => Number(item));
  let datesTwo = dateTwo.split("-").map((item) => Number(item));
  const diff = [0, 0, 0].map((value, index) => {
    return datesOne[index] - datesTwo[index];
  });
  return diff[0] * 12 + diff[1] + "月" + diff[2] + "天";
}

# 时间戳转化为年月日

/** * 时间戳转化为年月日 *
 * @param times 时间戳
 * @param ymd 格式类型(yyyy-mm-dd,yyyy/mm/dd)
 * @param hms 可选,格式类型(hh,hh:mm,hh:mm:ss)
 * @returns {年月日}
 */
export const timesToYyMmDd = (times, ymd, hms) => {
  const oDate = new Date(times);
  const oYear = oDate.getFullYear();
  const oMonth = oDate.getMonth() + 1;
  const oDay = oDate.getDate();
  const oHour = oDate.getHours();
  const oMin = oDate.getMinutes();
  const oSec = oDate.getSeconds();
  let oTime; // 最后拼接时间
  // 年月日格式
  switch (ymd) {
    case "yyyy-mm-dd":
      oTime = oYear + "-" + getzf(oMonth) + "-" + getzf(oDay);
      break;
    case "yyyy/mm/dd":
      oTime = oYear + "/" + getzf(oMonth) + "/" + getzf(oDay);
      break;
  }
  // 时分秒格式
  switch (hms) {
    case "hh":
      oTime = " " + oTime + getzf(oHour);
      break;
    case "hh:mm":
      oTime = oTime + getzf(oHour) + ":" + getzf(oMin);
      break;
    case "hh:mm:ss":
      oTime = oTime + getzf(oHour) + ":" + getzf(oMin) + ":" + getzf(oSec);
      break;
  }
  return oTime;
};

# 生成一周时间

/**
 * @function: 生成一周时间
 * @description: new Array 创建的数组只是添加了length属性，并没有实际的内容。通过扩展后，变为可用数组用于循环

export function getWeekTime() {
  return [...new Array(7)].map((j, i) =>
    new Date(Date.now() + i * 8.64e7).toLocaleDateString()
  );
}

# 日期格式化

/**
 * @function 日期格式化
 * @description 一个很灵活的日期格式化函数，可以根据使用者给定的格式进行格式化，能应对大部分场景
 * @param {string} format
 * @param {number} timestamp - 时间戳
 * @return {string}
 */
export function formatDate(format = "Y-M-D h:m", timestamp = Date.now()) {
  let date = new Date(timestamp);
  let dateInfo = {
    Y: date.getFullYear(),
    M: date.getMonth() + 1,
    D: date.getDate(),
    h: date.getHours(),
    m: date.getMinutes(),
    s: date.getSeconds(),
  };
  let formatNumber = (n) => (n > 10 ? n : "0" + n);
  let res = format
    .replace("Y", dateInfo.Y)
    .replace("M", dateInfo.M)
    .replace("D", dateInfo.D)
    .replace("h", formatNumber(dateInfo.h))
    .replace("m", formatNumber(dateInfo.m))
    .replace("s", formatNumber(dateInfo.s));
  return res;
}
