# 删除

/**
 * splice()
 * 第一个参数（2）定义了应添加新元素的位置（拼接）。
 * 第二个参数（0）定义应删除多少元素
 * 其余参数（“Lemon”，“Kiwi”）定义要添加的新元素。
 * 结果：["Banana", "Orange", "Lemon", "Kiwi"]
 */
let arr = ["Banana", "Orange", "Apple", "Mango"];
arr.splice(2, 2, "Lemon", "Kiwi");

/**
 * 使用delete 运算符来删除
 * 打印出的索引为1,2,3   arr 中的首个元素改为arr[0] undefined
 */
let arr = ["Banana", "Orange", "Apple", "Mango"];
delete arr[0]; // 把 arr 中的首个元素改为 undefined
/**
 * 从数组中删除最后一个元素
 * 从 arr 删除最后一个元素（"Mango"）
 * 结果：["Banana", "Orange", "Apple"]
 */
let arr = ["Banana", "Orange", "Apple", "Mango"];
arr.pop();

/**
 * 从数组中删除第一个元素
 * 从 arr 删除第一个元素（"Banana"）
 * 结果：["Orange", "Apple", "Mango"]
 */
let arr = ["Banana", "Orange", "Apple", "Mango"];
arr.shift();

# 获取

/**

/**
 * 从数组中删除第一个元素，并获取
 * 从 arr 获取第一个元素（"Mango"）

 */

let a = arr.shift();

# 遍历数组

/**
 * some() 方法用于检测数组中的元素是否满足指定条件（函数提供）。
 * some() 方法会依次执行数组的每个元素：
 * 参数1：currentValue   必须。当前元素的值
 * 参数2：index          选。当前元素的索引值
 * 参数3：arr 	         可选。数组本身
 * 参数4：thisValue      可选。对象作为该执行回调时使用，传递给函数，用作 "this" 的值。 如果省略了 thisValue ，"this" 的值为 "undefined"
 * 返回值：布尔值。如果数组中有元素满足条件返回 true，否则返回 false。
 * 输出结果 true
 */
let ages = [3, 10, 18, 20];
function checkAdult(age) {
  return age >= 18;
}
let a = ages.some(checkAdult);

/**
 * 参数1：currentValue 当前项（指遍历时正在被处理那个数组项）的值。
 * 参数2：index 当前项的索引（或下标）。
 * 参数3：array 数组本身。
 * 参数4：thisArg 可选参数。用来当作callback 函数内this的值的对象。
 * forEach 方法按升序为数组中含有效值的每一项执行一次callback 函数，
 * 那些已删除（使用delete方法等情况）或者从未赋值的项将被跳过（但不包括哪些值为undefine的项）。
 */
let a = "";
let arr = [45, 4, 9, 16, 25];
arr.forEach(myFunction);

function myFunction(value) {
  a = a + value;
}

/**
 * 通过对每个数组元素执行函数来创建新数组。
 * 不会对没有值的数组元素执行函数。
 * 不会更改原始数组
 * 参数1：currentValue 当前项（指遍历时正在被处理那个数组项）的值。
 * 参数2：index 当前项的索引（或下标）。
 * 参数3：array 数组本身。
 * 参数4：thisArg 可选参数。用来当作callback 函数内this的值的对象。
 */
let arr1 = [45, 4, 9, 16, 25];
let arr2 = arr1.map(myFunction);

function myFunction(value, index, array) {
  return value * 2;
}

/**
 * 创建一个包含通过测试的数组元素的新数组。
 * 参数1：currentValue 当前项（指遍历时正在被处理那个数组项）的值。
 * 参数2：index 当前项的索引（或下标）。
 * 参数3：array 数组本身。
 * 参数4：thisArg 可选参数。用来当作callback 函数内this的值的对象。
 */
let arr = [45, 4, 9, 16, 25];
let over18 = arr.filter(myFunction);

function myFunction(value, index, array) {
  return value > 18;
}

# 数组中的元素相加

/**
 * reduce() 方法在每个数组元素上运行函数，以生成（减少它）单个值。
 * reduce() 方法不会减少原始数组。
 * reduce() 方法在数组中从左到右工作。
 * 参数1：总数（初始值/先前返回的值）
 * 参数2：项目值
 * 参数3：项目索引
 * 参数4：数组本身
 */
let arr = [45, 4, 9, 16, 25];
let sum = arr.reduce(myFunction);

function myFunction(total, value, index, array) {
  return total + value;
}

# 搜索元素值并返回其位置

/**

*/
let arr = ["Apple", "Orange", "Apple", "Mango"];
let a = arr.indexOf("Apple");

/**
 * 数组结尾开始搜索。
*/
let arr = ["Apple", "Orange", "Apple", "Mango"];
let a = arr.lastIndexOf("Apple");
